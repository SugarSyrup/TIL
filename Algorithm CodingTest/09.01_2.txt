function solution(no, works) {
    works.sort((a,b) => b-a);
    
    for(var i = 0; i<no; i++){
        var work = works.shift();
        works.push(--work);
        works.sort((a,b) => b-a);
    }
    
    works[0] = works[0] * works[0];
    return works.reduce((p,c) => p+c*c);

//배상비용 최소화
}
